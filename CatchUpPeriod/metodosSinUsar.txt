         public List<Book> getCategoryBooks(String category) {
                 if(category == null || category.isEmpty()) {
                     return null;
                 }
                 List<Book> result = booksInterface.findByCategory_Code(category);
                 for (Book book : result) {
                     int activos = lendingRepository.countByBookAndReturningdateIsNull(book);
                     int disponibles = book.getCopies() - activos;

                     book.setCopies(disponibles);
                 }
                 return result;
             }
