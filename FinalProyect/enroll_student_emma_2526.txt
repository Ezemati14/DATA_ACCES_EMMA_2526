-- Función para matricular un estudiante en un curso
CREATE OR REPLACE FUNCTION _da_vtschool_2526.enroll_student_emma_2526(
    p_student_id VARCHAR,
    p_course_id INT
)
RETURNS TABLE(subject_id INT, subject_name VARCHAR)
LANGUAGE plpgsql
AS $$
DECLARE
    v_exists_student INT;
    v_exists_course INT;
    v_pending_count INT;
    v_enroll_code INT;
BEGIN
    -- Validar existencia de estudiante
    SELECT COUNT(*) INTO v_exists_student
    FROM _da_vtschool_2526.students
    WHERE idcard = p_student_id;

    IF v_exists_student = 0 THEN
        RAISE EXCEPTION 'ERROR: El estudiante % no existe.', p_student_id;
    END IF;

    -- Validar existencia de curso
    SELECT COUNT(*) INTO v_exists_course
    FROM _da_vtschool_2526.courses
    WHERE code = p_course_id;

    IF v_exists_course = 0 THEN
        RAISE EXCEPTION 'ERROR: El curso % no existe.', p_course_id;
    END IF;

    -- Verificar si ya tiene matrícula en este curso (scores históricos)
    SELECT COUNT(*) INTO v_pending_count
    FROM _da_vtschool_2526.scores s
    JOIN _da_vtschool_2526.enrollments e ON s.enrollment_id = e.code
    WHERE e.student = p_student_id
      AND e.course = p_course_id;

    -- Crear nueva matrícula en enrollments
    INSERT INTO _da_vtschool_2526.enrollments(student, course, year)
    VALUES (p_student_id, p_course_id, EXTRACT(YEAR FROM CURRENT_DATE)::int)
    RETURNING code INTO v_enroll_code;

    IF v_pending_count = 0 THEN
        -- Primera matrícula → asignaturas de primer curso
        RETURN QUERY
        SELECT s.code, s.name
        FROM _da_vtschool_2526.subjects s
        JOIN _da_vtschool_2526.subject_courses sc ON sc.subject_id = s.code
        WHERE sc.course_id = p_course_id
          AND s.year = 1
        ORDER BY s.code;

        -- Insertar en scores
        INSERT INTO _da_vtschool_2526.scores(enrollment_id, subject_id)
        SELECT v_enroll_code, s.code
        FROM _da_vtschool_2526.subjects s
        JOIN _da_vtschool_2526.subject_courses sc ON sc.subject_id = s.code
        WHERE sc.course_id = p_course_id
          AND s.year = 1;

    ELSE
        -- Matrícula previa → asignaturas pendientes
        SELECT COUNT(*) INTO v_pending_count
        FROM _da_vtschool_2526.subjects_pending_emma_2526(p_student_id, p_course_id);

        IF v_pending_count = 0 THEN
            -- No quedan pendientes → error
            RAISE EXCEPTION 'ERROR: El estudiante % ya completó todas las asignaturas del curso % y no puede matricularse de nuevo.', p_student_id, p_course_id;
        END IF;

        -- Devolver asignaturas pendientes
        RETURN QUERY
        SELECT s.subject_id, sub.name
        FROM _da_vtschool_2526.subjects_pending_emma_2526(p_student_id, p_course_id) s
        JOIN _da_vtschool_2526.subjects sub ON sub.code = s.subject_id
        ORDER BY s.subject_id;

        -- Insertar en scores
        INSERT INTO _da_vtschool_2526.scores(enrollment_id, subject_id)
        SELECT v_enroll_code, s.subject_id
        FROM _da_vtschool_2526.subjects_pending_emma_2526(p_student_id, p_course_id) s;
    END IF;
END;
$$;
